def human_size(num_bytes: int) -> str:
    size = float(num_bytes)
    for unit in ("B", "KB", "MB", "GB"):
        if size < 1024:
            return f"{size:.1f} {unit}"
        size /= 1024
    return f"{size:.1f} TB"


def detect_algorithm(h: str):
    return {32: "MD5", 40: "SHA-1", 64: "SHA-256"}.get(len(h))
